# 1.图片显示

小程序模拟器能显示，安卓机和真机调试不行，查了半天还是名字，空格，和资源地址的问题
原图片地址如下

src='../../assets/image/上拉.png'
1

首先相对地址，改为绝对地址，其二不要中文，其三""中不要有空格，就解决好了，更改为

src='/assets/image/up.png'
1

# 3.数组筛选

从后台获取数据后如果后台未进行筛选，则筛选工作交由前台

数组

  • filter:可以筛选指定含有指定元素的数组
  • map：可以筛选数组的指定元素

替换

  • 暂时考虑用for循环的方法，但是会很慢
  • 拼接的方法依旧可以用于循环
getlogs: util.debounce(function () {
var that = this
//优化结构，将请求到的数据存入缓存，便于第二次快速调用
var value = wx.getStorageSync('loglist')
if (!value) {
    console.log('请求日志')
    http.getRequest('/monitor/operlog/list', '', function (res) {
    console.log(res);
    that.setData({
        log: res.rows
    })
    //拿去数据后进行第一次刷选，缩小遍历数组大小
    var log = that.data.log;
    let loglist = log.filter(res => {
        return res.title == '设备详情'
    })
    //存入新数组，便于接下来的操作
    that.setData({
        loglist: loglist
    })
    //由于要进行字符串转换，用map将要转换的部分提取
    let new_params = loglist.map(obj => {
        return obj.operParam;
    })
    //循环数组，将数组中的每个对象转换并赋值给新的数组
    for (i = 0; i < new_params.length; i++) {
        var param = JSON.parse(new_params[i])
        var change = "loglist[" + i + "].operParam";
        that.setData({
        [change]: param
        })
    }
    wx.setStorageSync('loglist', that.data.loglist)
    //存入缓存，第一部分结束，
    //第二部分，拿取数据，并进行筛选，需要的数据，存入data，传递给前台渲染
    wx.getStorage({
        key: 'loglist',
        success() {
        let logs = loglist.filter(e => e.operParam.sn == that.data.details.sn)
        console.log('加载完成,输出日志:', logs)
        that.setData({
            logs: logs
        })
        }
    })
    that.setData({
        hasMore: false
    })
    })
} else {
//第二次进入，直接读取缓存，拿到数据后渲染到前台
    wx.getStorage({
    key: 'loglist',
    success(res) {
        let loglist = res.data
        let logs = loglist.filter(e => e.operParam.sn == that.data.details.sn)
        console.log('获取本地日志成功:', logs)
        that.setData({
        logs: logs
        })
    }
    })
    that.setData({
    hasMore: false
    })
}
},500),
