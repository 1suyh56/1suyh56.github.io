# utils.request网络请求封装

/**查看接口文档
 * host: 请求地址
 * url：接口
 * method：请求方法
 * postData：参数
 * doSuccess：成功的回调函数
 * doFail：失败的回调函数 
 * show作用：请求接口时，显示loadding，默认"false"
 */
/*
最简单的配置方式
* var app = getApp();
* var host = '接口名'; //接口地址
*/
//在小程序中可以利用开放接口配置不同环境下的接口
// 获取当前帐号信息
const accountInfo = wx.getAccountInfoSync();
// env类型
export const env = accountInfo.miniProgram.envVersion;
if (!env) {
  console.error("获取运行环境失败!");
}
const baseApi = {
  // 开发版
  develop: "http://192.168******************",
  // 体验版
  trial: "https://xxx",
  // 正式版
  release: "https://xxxx"
};

// request请求baseURL
export const host = baseApi[env];

function request(url, postData, method, doSuccess, doFail, show) {
  console.log("接口名:" + host + url);
  console.log("接口参数：" + JSON.stringify(postData)); //转换为json字符串
  var token = wx.getStorageSync('token');
  console.log("Authorization:bearer" + token); //token必须使用 Bearer Token(Token令牌)方式,使用该方式头部会多一个 authorization。
  var head;

  if (!token) {
    head = { //请求头
      "Content-Type": "application/json;charset=UTF-8"
    }
  } else {
    head = {
      "Content-Type": "application/json;charset=UTF-8",
      'Authorization': 'Bearer ' + token //+wx.getStorageSync('token') var定义的token传入
    }
  }

  wx.request({ //将方法内的参数全部列入请求
    url: host + url, //请求地址
    method: method, //请求方法
    header: head,
    data: postData,
    dataType: 'json', //返回数据格式
    responseType: 'text', //响应的数据类型
    success: function (res) {
      //成功执行方法，参数为res.data，直接将数据传入
      console.log("res.data:", res.data);
      if (res.data.code == 200) {
        doSuccess(res.data)
        if (!show) {
          wx.showToast({
            title: res.data.msg,
            icon: "success",
            duration: 1500
          });
        }
      } 
    },
    fail: function (res) {
      doFail(res);
    },
  })
}

//POST请求
function post(url, postData, doSuccess, doFail, show) {
  request(url, postData, "POST", doSuccess, doFail, show);
}

//GET请求
function get(url, postData, doSuccess, doFail, show) {
  request(url, postData, "GET", doSuccess, doFail, show);
}

/**获取验证码 */
function getSms(postData, doSuccess, doFail) {
  request('sms', postData, "GET", doSuccess, doFail);
}

module.exports = {
  postRequest: post,
  getRequest: get,
  getSms: getSms,
}
