# 1. 防抖和节流：

防抖和节流都能控制事件相应的频率,防抖在表单提交等等只希望提交一次的场景使用，无论触发多久，只提交最后一次，避免重复提交。节流在搜索框等不停变化等场景出现，第一次触发后以后，n秒内不管触发多少次，都不执行。直到下一个单位时间n秒，第一次触发函数并执行，这个n秒内不管函数多少次都不执行。
思考：为什么在点赞等场景推荐使用节流？

/*节流*/
export function throttle(func,interval){
    var enterTime =0;//触发时间
    var gapTime = innterval || 300;//间隔时间，如果interval不传，则默认300ms
    return function(){
    var context =this;
    var backTime =new date();
    if(backTime - enterTime > gapTime) {
        func.call(context,arguments[0]);
        enterTime =backTime;//赋值给带一次触发时间，这样就保存了第二次触发的时间
    }
    };
}
/*防抖*/
export function debounce(func,interval){
    var timer;//触发时间
    var gapTime = innterval || 1000;//间隔时间，如果interval不传，则默认1000ms
    return function(){
    clearTimeout(timer)
    var context = this;
    var args = arguments[0];//保存此处的arguments，timeout是全局的，不是防抖函数需要的
    timer =setTimeout(function () {
        func.call(context,args);
    },gapTime)
    };
}

# 2. 判断数据类型

使用Object.prototype.toString.call()的方式来判断一个变量的类型是最准确的方法。

  • number ： 数值
  • string ：字符串
  • boolean：布尔值
  • object：对象
  • function：函数
  • array : 数组
  • date：日期
  • regexp：正则
